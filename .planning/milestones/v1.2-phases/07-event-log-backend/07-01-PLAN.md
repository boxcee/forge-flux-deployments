        ```sql
          SUM(CASE WHEN status_code = 200 THEN 1 ELSE 0 END) as accepted,
          SUM(CASE WHEN status_code IN (400, 401, 502, 503) THEN 1 ELSE 0 END) as failed,
          SUM(CASE WHEN status_code = 204 THEN 1 ELSE 0 END) as skipped
        FROM webhook_events
        WHERE timestamp > ?
        ```
